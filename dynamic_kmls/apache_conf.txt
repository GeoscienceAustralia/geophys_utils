# ensure Apache has mod_wsgi installed
sudo a2enmod wsgi

# lines to add to Apache conf (usually /etc/apache2/sites-available/000-default.conf):

        # KML Server Python Flask API
        WSGIDaemonProcess kmlapp threads=5
        WSGIScriptAlias /kml /var/www/html/geophys_utils/dynamic_kmls/app.wsgi

        <Directory kmlapp>
            WSGIProcessGroup kmlapp
            WSGIApplicationGroup %{GLOBAL}
            Require all denied
        </Directory>


# reload Apache
sudo service apache2 reload
